Table authors {
  id_author int [pk]
  name varchar
  email varchar
}

Table affiliation {
  id_affiliation int [pk]
  name varchar
  type varchar
  country varchar
  
}

Table affiliation_authors {
  id_affiliation int 
  id_author int
indexes{
  (id_affiliation,id_author) [pk]
}
}

Table articles {
  id_article int [pk]
  title varchar
  publication_date datetime
  abstract text
  studylink text
  id_journal int
  indexes {
    (id_journal) [pk]
  }
}

Table studytype {
  id_study int [pk]
  name varchar
}

Table sous_theme_articles {
  id_sous_theme int
  id_article int 
  indexes { (id_article,id_sous_theme) [pk] }
}

Table articles_authors {
  id_article int 
  id_author int
  indexes {
    (id_article,id_author)  [pk]
  }
  
}

Table journal {
  id_journal int [pk]
  name varchar
}


Table theme {
  id_theme int [pk]
  name varchar
  
}

Table sous_theme {
  id_sous_theme int [pk]
  id_theme int
  name text
  indexes {
    id_theme [pk]
  }
}

Table studytype_articles{
  id_study int
  id_article int
  indexes{
    (id_article,id_study) [pk]
  }
}


Ref : affiliation.id_affiliation < affiliation_authors.id_affiliation
Ref : authors.id_author < affiliation_authors.id_author
Ref : studytype.id_study < studytype_articles.id_study
Ref : articles.id_article < studytype_articles.id_article
Ref : articles.id_journal > journal.id_journal
Ref : authors.id_author < articles_authors.id_author
Ref : articles.id_article < articles_authors.id_article
Ref : theme.id_theme < sous_theme.id_theme
Ref : articles.id_article < sous_theme_articles.id_article
Ref : sous_theme.id_sous_theme < sous_theme_articles.id_sous_theme
